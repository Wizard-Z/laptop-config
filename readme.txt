to install sdkman
	-open terminal using (ALT+CTRL+T)
	-type curl -s "https://get.sdkman.io" | bash
	-press enter
	-type source "$HOME/.sdkman/bin/sdkman-init.sh"
	-press enter

to run sdk_install.sh
	- open terminal using (ALT+CTRL+T)
	-type chmod 777 sdk_install.sh
	-press enter
	-type ./sdk_install.sh
	-press enter
